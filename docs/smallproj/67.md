简单替换密码用一个字母代替另一个字母。由于字母 `A` 有 26 种可能的替换， `B` 有 25 种可能的替换， `C` 有 24 种可能的替换，等等，所以可能的键的总数是 26 × 25 × 24 × 23 ×。。。× 1，即 403291461126605635584000000 键！对于一台超级计算机来说，这对于暴力破解来说太多了，所以项目 7“凯撒黑客”中使用的密码破解方法不能用于对抗简单的密码。不幸的是，狡猾的攻击者可以利用已知的弱点来破解代码。如果你想了解更多关于密码和密码破解的知识，你可以阅读我的书[《Python 密码破解指南》](https://nostarch.com/crackingcodes/)（NoStarch出版社，2018）。

## 它是如何工作的

键的 26 个字母中的每一个的位置对应于字母表中相同位置的字母:

![f66001](img/e77603443ad0c92246d720899729fc58.png)

:字母表中的字母如何用一个以 `WNOM` 开头的密钥加密。若要解密，请将底部的字母替换为上面相应的字母。

用这个密钥，字母 `A` 加密到 `W` (而 `W` 解密到 `A` ，字母 `B` 加密到 `N` ，以此类推。`LETTERS`和`key`变量被分配给`charsA`和`charsB`(或者在解密时反过来)。用`charsB`中的相应字符替换`charsA`中的任何消息字符，以产生最终翻译的消息。

```py
"""Simple Substitution Cipher, by Al Sweigart email@protected #  1
A simple substitution cipher has a one-to-one translation for each #  2
symbol in the plaintext and each symbol in the ciphertext. #  3
More info at: https://en.wikipedia.org/wiki/Substitution_cipher #  4
This code is available at https://nostarch.com/big-book-small-python-programming #  5
Tags: short, cryptography, math""" #  6
 #  7
import random #  8
 #  9
try: #  10
   import pyperclip  # pyperclip copies text to the clipboard. #  11
except ImportError: #  12
   pass  # If pyperclip is not installed, do nothing. It's no big deal. #  13
 #  14
# Every possible symbol that can be encrypted/decrypted: #  15
LETTERS = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ' #  16
 #  17
def main(): #  18
   print('''Simple Substitution Cipher, by Al Sweigart #  19
A simple substitution cipher has a one-to-one translation for each #  20
symbol in the plaintext and each symbol in the ciphertext.''') #  21
 #  22
   # Let the user specify if they are encrypting or decrypting: #  23
   while True:  # Keep asking until the user enters e or d. #  24
       print('Do you want to (e)ncrypt or (d)ecrypt?') #  25
       response = input('> ').lower() #  26
       if response.startswith('e'): #  27
           myMode = 'encrypt' #  28
           break #  29
       elif response.startswith('d'): #  30
           myMode = 'decrypt' #  31
           break #  32
       print('Please enter the letter e or d.') #  33
 #  34
   # Let the user specify the key to use: #  35
   while True:  # Keep asking until the user enters a valid key. #  36
       print('Please specify the key to use.') #  37
       if myMode == 'encrypt': #  38
           print('Or enter RANDOM to have one generated for you.') #  39
       response = input('> ').upper() #  40
       if response == 'RANDOM': #  41
           myKey = generateRandomKey() #  42
           print('The key is {}. KEEP THIS SECRET!'.format(myKey)) #  43
           break #  44
       else: #  45
           if checkKey(response): #  46
               myKey = response #  47
               break #  48
 #  49
   # Let the user specify the message to encrypt/decrypt: #  50
   print('Enter the message to {}.'.format(myMode)) #  51
   myMessage = input('> ') #  52
 #  53
   # Perform the encryption/decryption: #  54
   if myMode == 'encrypt': #  55
       translated = encryptMessage(myMessage, myKey) #  56
   elif myMode == 'decrypt': #  57
       translated = decryptMessage(myMessage, myKey) #  58
 #  59
   # Display the results: #  60
   print('The %sed message is:' % (myMode)) #  61
   print(translated) #  62
 #  63
   try: #  64
       pyperclip.copy(translated) #  65
       print('Full %sed text copied to clipboard.' % (myMode)) #  66
   except: #  67
       pass  # Do nothing if pyperclip wasn't installed. #  68
 #  69
 #  70
def checkKey(key): #  71
   """Return True if key is valid. Otherwise return False.""" #  72
   keyList = list(key) #  73
   lettersList = list(LETTERS) #  74
   keyList.sort() #  75
   lettersList.sort() #  76
   if keyList != lettersList: #  77
       print('There is an error in the key or symbol set.') #  78
       return False #  79
   return True #  80
 #  81
 #  82
def encryptMessage(message, key): #  83
   """Encrypt the message using the key.""" #  84
   return translateMessage(message, key, 'encrypt') #  85
 #  86
 #  87
def decryptMessage(message, key): #  88
   """Decrypt the message using the key.""" #  89
   return translateMessage(message, key, 'decrypt') #  90
 #  91
 #  92
def translateMessage(message, key, mode): #  93
   """Encrypt or decrypt the message using the key.""" #  94
   translated = '' #  95
   charsA = LETTERS #  96
   charsB = key #  97
   if mode == 'decrypt': #  98
       # For decrypting, we can use the same code as encrypting. We #  99
        # just need to swap where the key and LETTERS strings are used. # 100
        charsA, charsB = charsB, charsA # 101
 # 102
    # Loop through each symbol in the message: # 103
    for symbol in message: # 104
        if symbol.upper() in charsA: # 105
            # Encrypt/decrypt the symbol: # 106
            symIndex = charsA.find(symbol.upper()) # 107
            if symbol.isupper(): # 108
                translated += charsB[symIndex].upper() # 109
            else: # 110
                translated += charsB[symIndex].lower() # 111
        else: # 112
            # The symbol is not in LETTERS, just add it unchanged. # 113
            translated += symbol # 114
 # 115
    return translated # 116
 # 117
 # 118
def generateRandomKey(): # 119
    """Generate and return a random encryption key.""" # 120
    key = list(LETTERS)  # Get a list from the LETTERS string. # 121
    random.shuffle(key)  # Randomly shuffle the list. # 122
    return ''.join(key)  # Get a string from the list. # 123
 # 124
 # 125
# If this program was run (instead of imported), run the program: # 126
if __name__ == '__main__': # 127
    main()  # 128
```
