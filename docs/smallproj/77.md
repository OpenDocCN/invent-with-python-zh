井字游戏是一种在 3 × 3 网格上玩的经典纸笔游戏。玩家轮流放置 X 或 O 标记，试图连续获得三个。大多数井字游戏都以平局告终，但如果你的对手不小心，你也有可能智胜他们。

## 它是如何工作的

为了在这个程序中表示井字游戏棋盘，我们使用了一个字典，用键`'1'`到`'9'`来表示棋盘上的空格。数字空间的排列方式与手机键盘相同。本词典中的值是代表球员标记的字符串`'X'`或`'O'`和代表空白的`' '`。

```py
"""Tic-Tac-Toe, by Al Sweigart email@protected #  1
The classic board game. #  2
This code is available at https://nostarch.com/big-book-small-python-programming #  3
Tags: short, board game, game, two-player""" #  4
 #  5
ALL_SPACES = ['1', '2', '3', '4', '5', '6', '7', '8', '9'] #  6
X, O, BLANK = 'X', 'O', ' '  # Constants for string values. #  7
 #  8
 #  9
def main(): #  10
   print('Welcome to Tic-Tac-Toe!') #  11
   gameBoard = getBlankBoard()  # Create a TTT board dictionary. #  12
   currentPlayer, nextPlayer = X, O  # X goes first, O goes next. #  13
 #  14
   while True:  # Main game loop. #  15
       # Display the board on the screen: #  16
       print(getBoardStr(gameBoard)) #  17
 #  18
       # Keep asking the player until they enter a number 1-9: #  19
       move = None #  20
       while not isValidSpace(gameBoard, move): #  21
           print('What is {}\'s move? (1-9)'.format(currentPlayer)) #  22
           move = input('> ') #  23
       updateBoard(gameBoard, move, currentPlayer)  # Make the move. #  24
 #  25
       # Check if the game is over: #  26
       if isWinner(gameBoard, currentPlayer):  # Check for a winner. #  27
           print(getBoardStr(gameBoard)) #  28
           print(currentPlayer + ' has won the game!') #  29
           break #  30
       elif isBoardFull(gameBoard):  # Check for a tie. #  31
           print(getBoardStr(gameBoard)) #  32
           print('The game is a tie!') #  33
           break #  34
       # Switch turns to the next player: #  35
       currentPlayer, nextPlayer = nextPlayer, currentPlayer #  36
   print('Thanks for playing!') #  37
 #  38
 #  39
def getBlankBoard(): #  40
   """Create a new, blank tic-tac-toe board.""" #  41
   # Map of space numbers: 1|2|3 #  42
   #                       -+-+- #  43
   #                       4|5|6 #  44
   #                       -+-+- #  45
   #                       7|8|9 #  46
   # Keys are 1 through 9, the values are X, O, or BLANK: #  47
   board = {} #  48
   for space in ALL_SPACES: #  49
       board[space] = BLANK  # All spaces start as blank. #  50
   return board #  51
 #  52
 #  53
def getBoardStr(board): #  54
   """Return a text-representation of the board.""" #  55
   return ''' #  56
{}|{}|{} 1 2 3 #  57
     -+-+- #  58
{}|{}|{} 4 5 6 #  59
     -+-+- #  60
{}|{}|{} 7 8 9'''.format(board['1'], board['2'], board['3'], #  61
                               board['4'], board['5'], board['6'], #  62
                               board['7'], board['8'], board['9']) #  63
 #  64
def isValidSpace(board, space): #  65
   """Returns True if the space on the board is a valid space number #  66
   and the space is blank.""" #  67
   return space in ALL_SPACES and board[space] == BLANK #  68
 #  69
 #  70
def isWinner(board, player): #  71
   """Return True if player is a winner on this TTTBoard.""" #  72
   # Shorter variable names used here for readablility: #  73
   b, p = board, player #  74
   # Check for 3 marks across 3 rows, 3 columns, and 2 diagonals. #  75
   return ((b['1'] == b['2'] == b['3'] == p) or  # Across top #  76
           (b['4'] == b['5'] == b['6'] == p) or  # Across middle #  77
           (b['7'] == b['8'] == b['9'] == p) or  # Across bottom #  78
           (b['1'] == b['4'] == b['7'] == p) or  # Down left #  79
           (b['2'] == b['5'] == b['8'] == p) or  # Down middle #  80
           (b['3'] == b['6'] == b['9'] == p) or  # Down right #  81
           (b['3'] == b['5'] == b['7'] == p) or  # Diagonal #  82
           (b['1'] == b['5'] == b['9'] == p))    # Diagonal #  83
 #  84
def isBoardFull(board): #  85
   """Return True if every space on the board has been taken.""" #  86
   for space in ALL_SPACES: #  87
       if board[space] == BLANK: #  88
           return False  # If any space is blank, return False. #  89
   return True  # No spaces are blank, so return True. #  90
 #  91
 #  92
def updateBoard(board, space, mark): #  93
   """Sets the space on the board to mark.""" #  94
   board[space] = mark #  95
 #  96
 #  97
if __name__ == '__main__': #  98
   main()  # Call main() if this module is run, but not when imported.  #  99
```
